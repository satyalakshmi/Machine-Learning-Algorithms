# Using the elbow method to find the optimal number of clusters
from sklearn.cluster import KMeans# importing the KMeans from sklearn
wcss = [] # list for wcss
for i in range(1, 11): # for loop for the sum of sqaures
    kmeans = KMeans(n_clusters = i, init = 'k-means++', random_state = 42)# creating object for KMeans class with some parameters
    #Here init = random initiliazation which is choosen as K-means++
    kmeans.fit(X) # fitting the Kmeans classifier object to the X data
    wcss.append(kmeans.inertia_) # computes the within sum of squares and append to the wcss list taken 
plt.plot(range(1, 11), wcss)# plotting the wcss list
plt.title('The Elbow Method')# title for the entire plot
plt.xlabel('Number of clusters')# title for the x-axis in the plot
plt.ylabel('WCSS')# title for the y-axis in the plot
plt.show()
