# Visualising the Training set results
import warnings
warnings.filterwarnings('ignore')
from matplotlib.colors import ListedColormap # this class helps us to plot the colourized data points
X_set, y_set = X_train, y_train # replacing the training variables with some local variables
X1, X2 = np.meshgrid(np.arange(start = X_set[:, 0].min() - 1, stop = X_set[:, 0].max() + 1, step = 0.01),
                     np.arange(start = X_set[:, 1].min() - 1, stop = X_set[:, 1].max() + 1, step = 0.01))
# here we took the grid type plot with min and max values of age and salary with a step limit of 0.01
plt.contourf(X1, X2, classifier.predict(np.array([X1.ravel(), X2.ravel()]).T).reshape(X1.shape),
             alpha = 0.75, cmap = ListedColormap(('red', 'green')))
# here we use the contourf function for getting the line between the predicted regions and therefore the 0's will 
# be plotted as red ones and 1's will be plotted as green ones in the plot
plt.xlim(X1.min(), X1.max())
plt.ylim(X2.min(), X2.max())
for i, j in enumerate(np.unique(y_set)):
    plt.scatter(X_set[y_set == j, 0], X_set[y_set == j, 1],
                c = ListedColormap(('red', 'green'))(i), label = j)
plt.title('Logistic Regression (Training set)') # given title of the plot
plt.xlabel('Age')# given the title for the x-axis
plt.ylabel('Estimated Salary')# given the title for the y-axis
plt.legend()
plt.show()
